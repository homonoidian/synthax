    macro capture(var) #

    Creates a capture with the same name as the given var (which should be the name of a variable holding the captured rule).

    boolean = keep("true" | "false", "literal")
null = "null"
value = capture(boolean) | capture(null)

"true".apply!(value)
# => root ⸢0-4⸥
#      boolean ⸢0-4⸥ {"literal" => "true"}

"false".apply!(value)
# => root ⸢0-5⸥
#      boolean ⸢0-5⸥ {"literal" => "false"}

"null".apply!(value)
# => root ⸢0-4⸥
#      null ⸢0-4⸥

    def maybe(exp) : Rule #

    Returns a rule that will match if exp (see Rule.from) matches zero or one times.

    xyzzy = "xy" & maybe('z') & "zy"

"xyzy".apply?(xyzzy)  # => Tree
"xyzzy".apply?(xyzzy) # => Tree
"xyy".apply?(xyzzy)   # => nil

    def sep(exp, *, by sexp) #

    Returns a rule that will match a list of one or more exps (see Rule.from) separated by sexps (same).

    foo = "foo"
foos = sep(foo, by: some(' ') & "and" & some(' '))

"".apply?(foos)                    # => nil
"foo".apply?(foos)                 # => Tree
"foo and foo".apply?(foos)         # => Tree
"foo and foo and foo".apply?(foos) # => Tree
